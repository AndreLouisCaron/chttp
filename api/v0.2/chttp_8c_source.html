102 static int _push_field (http_head * self, const char * field, size_t size)
103 {
104  size_t used = 0;
105  // Check that enough space is remaining.
106  if ((self->size-self->used-3) < size) {
107  return 0;
108  }
109  // Copy data.
110  while ((used < size) && (field[used] != '\0')) {
111  self->data[self->used++] = field[used++];
112  }
113  // Add null terminator.
114  self->data[self->used] = '\0';
115  return 1;
116 }

161 static int _commit (http_head * self, size_t mark)
162 {
163  size_t nulls = 0;
164  // Validate the mark.
165  if (mark >= (self->size-3)) {
166  return 0;
167  }
168  // Verify that the partial operations put valid null terminators.
169  while (mark <= self->used) {
170  nulls += (self->data[mark++] == '\0');
171  }
172  if ((nulls != 2) || (self->data[self->used] != '\0')) {
173  return 0;
174  }
175  // Restore buffer invariant.
176  self->data[self->used++] = '\0';
177  return 1;
178 }

222 int http_cursor_next (http_cursor * self)
223 {
224  char * text = self->head->data + self->base;
225  // Guard against empty head & extra iterations.
226  if (*text == '\0') {

228  return (0);
