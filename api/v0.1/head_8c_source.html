35 static size_t push_segment (char * data, const char * text, size_t size)
36 {
37  size_t used = 0;
38  while ((used < size-2) && (*text != '\0')) {
39  *data++ = *text++, ++used;
40  }
41  if (*text != '\0') {
42  // Out of space!
43  }
44  *data++ = '\0', ++used;
45  *data++ = '\0'; // Final null terminator (not counted).
46  return (used);
47 }

49 static size_t next_segment (char ** segment)
50 {
51  size_t used = 0;
52  while (**segment != '\0') {
53  ++*segment, ++used;
54  }
55  ++*segment, ++used;
56  return (used);
57 }

102 {

104  // Guard against empty head & extra iterations.
105  if (*text == '\0') {
